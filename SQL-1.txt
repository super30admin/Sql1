CREATE FUNCTION getNthHighestSalary(N INT) RETURNS INT
BEGIN
  RETURN (
    WITH CTE AS(SELECT *,dense_rank() OVER(ORDER BY salary DESC) as rnk
    FROM employee)
    SELECT DISTINCT IFNULL(null,salary)
    FROM CTE 
    WHERE rnk = N
  );
END
